def annotate_anchors(anchors_df, annon_df, column_search, column_name=None, search_point_bool=False, get_num_hits=False):
    
    """
    anchors_df must have columns named: 'chr', 'start', 'end'
    annon_df must have columns named: 
                if search_point_bool: <column_search>, 'chr', and 'loc' (extra columns are also ok)
                if not search_point_bool: <column_search>, 'chr', 'start', 'end'
                
    Note: both 'chr' columns in the dataframes must have 'chr' prefix
    search_point_bool = looking for single point or looking for regional overlap,
        search_point_bool true for snp, tss
                         false for atac, footprinting
    
    column_search <str> within annon_df
    column_name (optional Nonetype) or <str> that is the new column created in anchor df
    get_num_hits (optional <bool>, default False): whether or not to count the number of hits                  
                         
    sets a new column (column_name) of str type in anchors_df with a str of possible values separated by "::"
    """
    if column_name is None:
        column_name = column_search
    anchors_df[column_name]=''
    for idx, row in anchors_df.iterrows():
        if search_point_bool: # single 
            anchors_df.at[idx, column_name] = searchPoint(row, annon_df, column_search)
        else: # 
            anchors_df.at[idx, column_name] = searchOverlap(row, annon_df, column_search)
    
    if get_num_hits:
        func_count = lambda x: len(x.split('::')) if x else 0
        anchors_df['num_'+column_name] = anchors_df[column_name].apply(func_count)
    return anchors_df

def searchPoint(row, annon_df, column):
    annon_df = annon_df[annon_df.chr == row['chr']]
    annon_df = annon_df[(annon_df['loc'] <= int(row['end'])) & (annon_df['loc'] >= int(row['start']))]
    return "::".join(list(set(annon_df[column])))

def searchOverlap(row, annon_df, column,min_overlap=0.5):
    # must row region must contain at least some percentage of annotated region
    s_start, s_end = int(row['start']), int(row['end'])
#     s_buffer = (s_end - s_start)/2
    # filter by chromosome
    annon_df = annon_df[annon_df.chr == row['chr']]
    
    def get_perc_overlap_func(x):
        # x = start,end
        return percent_overlap(x,[s_start,s_end])
        
    if(annon_df.shape[0]>0):
        annon_df['perc_overlap'] = annon_df[['start','end']].apply(get_perc_overlap_func,axis=1)
        annon_df = annon_df[annon_df.perc_overlap > min_overlap]


        return '::'.join(list(set(annon_df[column])))
    else:
        return ''

def overlaps(a, b):
    """
    Return the amount of overlap, in bp
    between a and b.
    where a and b are tuples of size 2 consisting of [start,end] points, edges inclusive
    If >0, the number of bp of overlap
    If 0,  they are book-ended.
    If <0, the distance in bp between them
    """

    return min(a[1], b[1]) - max(a[0], b[0])

def percent_overlap(a,b):
    """
    returns the percentage of overlap between a and b
    
    """
    max_overlap = min(a[1]-a[0],b[1]-b[0])
    return max(0,overlaps(a,b)/float(max_overlap))

def footprinting_to_df(footprinting_file):
    footprinting_df = pd.read_table(footprinting_file,header=None)
    footprinting_df.columns = ['chr', 'start', 'end', 'motif', 'score', 'strand', 'other']
    temp = footprinting_df.motif.str.split('(') #i.e. remove (var.2)
    temp = temp.apply(lambda x: x[0])
    temp = temp.str.split('.')
    footprinting_df['motif_abbr'] = temp.apply(lambda x: x[-1])
    return footprinting_df

def atac_to_df(atac_narrowPeak_file):
    atac_df = pd.read_table(atac_narrowPeak_file,header=None)
    atac_df.columns = ['chr', 'start', 'end']
    atac_df = atac_df.groupby(atac_df.columns.tolist()).size().reset_index().rename(columns={0:'count'})
    atac_df['region'] = atac_df.apply(lambda row: '_'.join(row.values.astype(str)), axis=1)
    return atac_df

def hichip_to_df(hichip_file):
    with open(hichip_file, 'r') as f:
        hichip_dict = {}
        for idx, line in enumerate(f.readlines()):
            line_arr = line.strip().split()
    #         print(line_arr)
            if line_arr[0].startswith('chr'):
                prefix=''
            else:
                prefix='chr'
            start_A, end_A = line_arr[1], line_arr[2]
            start_B, end_B = line_arr[4], line_arr[5]
            if all([x.isdigit() for x in [start_A, end_A , start_B, end_B ]]):
                start_node=prefix+ line_arr[0] + '_' + start_A + '_' + end_A
                end_node = prefix+ line_arr[3] + '_' + start_B + '_' + end_B
                edge_attr = line_arr[7]
                hichip_dict[idx] = {'source': start_node, 'target':end_node, 'count':edge_attr}
        hichip_df = pd.DataFrame.from_dict(hichip_dict, orient='index')
#         
    return hichip_df

def hichip_to_anchor(hichip_df, sample):
    anchors_list = sorted(set(hichip_df.source).union(set(hichip_df.target)))
    anchors_df = pd.DataFrame({"anchors":anchors_list})
    anchors_df_split = anchors_df.anchors.str.split('_', expand=True)
    anchors_df_split.columns = ['chr', 'start', 'end']
    anchors_df = pd.concat([anchors_df, anchors_df_split], axis=1)
    anchors_df['sample']= sample
    return anchors_df
